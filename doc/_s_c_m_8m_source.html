619  function integerexflag = greedyLoop(double tol,integer maxIter) {
620 
621 
622  /* final setup for the greedy algorithm */
623  isDone = false;
624  exflag = 0;
625  curLoopCtr = 1;
626 
627  obj.L.info(" SCM ", ...
628  sprintf(" Starting greedy loop with %d max iterations and tolerance %.2e ", ...
629  maxIter, tol));
630 
631  /* Now we can finally start the greedy loop */
632  while ~isDone
633  scmtic = tic;
634  /* now we compute the relative gaps of upper and lower bound for all the
635  * parameters given in the training set to find the best choice for the
636  * next loop cycle */
637  reLBStability = zeros(obj.nMuTrain, 1);
638  reLBPositivity = zeros(obj.nMuTrain, 1);
639 
640  textprogressbar(" checking parameters: ");
641  for idx = 1:obj.nMuTrain
642  textprogressbar(idx * 100 / obj.nMuTrain);
643  if ~obj.offMuChosen(idx)
644  param = obj.offMuTrain(:, idx);
645 
646  /* set up the needed sets for the positivity constraint calculation */
647  paramsNotInCkPrev = true(obj.nMuTrain, 1);
648  paramsNotInCkPrev(obj.offMuCkIndex(1:(end-1))) = false;
649  paramsNotInCkPrevIndexes = obj.MuIndexes(paramsNotInCkPrev);
650 
651  /* get the nearest parameters for stability and positivity constraints */
652  [~, Idxalpha] = obj.getNeighbors(obj.Malpha, param, obj.offMuCk);
653  [~, IdxplusXi] = obj.getNeighbors(obj.Mplus, param, obj.offMuTrain(:, paramsNotInCkPrev));
654  Idxplus = paramsNotInCkPrevIndexes(IdxplusXi);
655 
656  /* check whether the lower bound could change */
657  reLBStability(idx) = any(obj.offMuCkIndex(Idxalpha) == obj.muCurIdx);
658  reLBPositivity(idx) = any(Idxplus == obj.muCurIdx);
659 
660  /* if one of the above conditions holds, then we compute new
661  * (hopefully better) lower bounds, if not, there is no improvement
662  * and we take the value from the last loop */
663  if reLBStability(idx) || reLBPositivity(idx)
664  [lb, ub] = obj.onlineQuery(param, true);
665  obj.muTrainLB(idx) = lb;
666  obj.muTrainUB(idx) = ub;
667  else
668  obj.muTrainLB(idx) = obj.offMuLB(idx);
669  obj.muTrainUB(idx) = min(obj.offYstarCk.^t * obj.offMuTrainMapped(:, idx));
670  end
671  end
672  end
673  textprogressbar(" done! ");
674 
675  /* save the newly calculated lower bounds for online use */
676  obj.offMuLB= obj.muTrainLB;
677 
678  if obj.save_scm_bounds
679  obj.scm_lower_bounds[obj.totalLoopCtr + 1] = sqrt(obj.muTrainLB);
680  obj.scm_upper_bounds[obj.totalLoopCtr + 1] = sqrt(obj.muTrainUB);
681  end
682 
683  /* Look for the parameter with the greatest gap / error */
684  obj.muPrevIdx= obj.muCurIdx;
685  [maxGap, obj.muCurIdx] = max((obj.muTrainUB - obj.muTrainLB) ./ obj.muTrainUB);
686  if ~isempty(maxGap)
687  obj.errors(end + 1) = maxGap;
688  end
689 
690  /* first check whether we are done here, check if the given tolerance
691  * is obtained */
692  if maxGap < tol
693  exflag = 1;
694  isDone = true;
695  obj.L.info(" SCM ", sprintf(" stopped greedy loop, max gap %f is smaller than tolerance %f! ", maxGap, tol));
696  break;
697  end
698 
699  /* looks like we're not done yet, so let's solve the eigenvalue
700  * problem. assemble the system for the chosen parameter and calculate
701  * the smallest eigenvalue and the corresponding eigenvector */
702  M = obj.assembleAffineT(obj.offMuTrainMapped(:, obj.muCurIdx));
703  [mi, mivec] = obj.computeEV(M, obj.normX, 0);
704  /* if the eigenvalue mi is positive, then we got the largest eigenvalue
705  * and have to do it again with a shift */
706  if mi >= 0
707  [mi, mivec] = obj.computeEV(M, obj.normX, mi);
708  end
709 
710  /* the new eigenvalue is lower and upper bound for this parameter mu */
711  obj.muTrainLB(obj.muCurIdx) = mi;
712  obj.muTrainUB(obj.muCurIdx) = mi;
713 
714  /* calculate the values of ystar for the current parameter. this is done
715  * by iterating over all the addends of the affine decomposition and
716  * multiplication with the eigenvector from above from both sides */
717  ystar = zeros(obj.nQt, 1);
718  for idx = 1:obj.nQt
719  ystar(idx) = mivec.^t * obj.affineT[idx] * mivec;
720  end
721 
722  /* save the newly generated data for the online stage */
723  obj.offMuCk(:, end + 1) = obj.offMuTrain(:, obj.muCurIdx);
724  obj.offMuCkMapped(:, end + 1) = obj.offMuTrainMapped(:, obj.muCurIdx);
725  obj.offMuCkIndex(end + 1) = obj.muCurIdx;
726  obj.offYstarCk(:, end + 1) = ystar;
727  obj.offAlphaCk(end + 1) = mi;
728  obj.offMuChosen(obj.muCurIdx) = true;
729 
730  /* logging */
731  obj.L.info(" SCM ", sprintf(" cycle %7.d of max %7.d done. max gap %g ", ...
732  curLoopCtr, maxIter, maxGap));
733 
734  /* update the current and total loop count */
735  curLoopCtr = curLoopCtr + 1;
736  obj.totalLoopCtr= obj.totalLoopCtr + 1;
737 
738  /* and now we check some more breaking conditions */
739  if curLoopCtr > maxIter
740  /* we stop because we performed the maxIter number of greedy cycles.
741  * it's now maybe a good idea to check maxGap and decide whether the
742  * offline stage should be restarted with a larger maxIter */
743  exflag = 2;
744  isDone = true;
745  obj.L.info(" SCM ", sprintf(" stopped greedy loop, number of max iterations %d reached! ", maxIter));
746  break;
747  elseif size(obj.offMuCk, 2) == obj.nMuTrain
748  /* we stop because the training set is empty. this really shouldn't
749  * happen if we want good bounds... */
750  exflag = 3;
751  isDone = true;
752  obj.L.info(" SCM ", " stopped greedy loop, training set is already empty! ");
753  break;
754  end
755  toc(scmtic)
756  end
757  }

852  function [doublemi , doublema , retries ] = computeMinMaxEV(matrix A,matrix B) {
853 
854 
855  /* first we calculate the eigenvalue with the largest absolute value */
856  [lm, ~, retries1] = obj.computeEV(A, B, 0);
857  /* if it's positive, then it's the maximal eigenvalue, else the
858  * minimal eigenvalue */
859  if lm >= 0
860  ma = lm;
861  /* now shift to get the minimal eigenvalue */
862  [mi, ~, retries2] = obj.computeEV(A, B, lm);

867  end
868

883  function [doublelm , colveclmvec , retries ] = computeEV(matrix A,matrix B,double shift) {
884 
885 
886  retries = 0;
887 
888  /* change the `eigenvalue won't converge` warning to an error, so that we
889  * can catch it. */
890  origstate = warning;
891  warning(" error ", " MATLAB:eigs:NoEigsConverged "); /* #ok<CTPCT> */
892 
893 
894  /* setup the options structure */
895  opts = struct();
896  opts.tol= eps;
897 
898  try
899  /* we first try to solve the eigenvalue problem with default settings. */
900  [lmvec, lm] = eigs(A - shift * B, B, 1, " lm ");
901  /* revert the shift */
902  lm = lm + shift;
903  catch err
904  /* if the the default settings won't work, we have to try again with some
905  * modifications */
906  if strcmp(err.identifier, " MATLAB:eigs:NoEigsConverged ") || strcmp(err.identifier, " MATLAB:eigs:ARPACKroutineErrorMinus14 ")
907  /* flag to check whether we found a eigenvalue */
908  hasEV = false;
909  /* decrease the accuracy */
910  opts.tol= opts.tol * 100;
911  /* set the number of used Lanczos vectors */
912  opts.p= 50;
913  while ~hasEV
914  try
915  retries = retries + 1;
916  /* and now try again */
917  [lmvec, lm] = eigs(A - shift * B, B, 1, " lm ", opts);
918  lm = lm + shift;
919  hasEV = true;
920  catch err2
921  hasEV = false;
922  if strcmp(err2.identifier, " MATLAB:eigs:NoEigsConverged ") || strcmp(err2.identifier, " MATLAB:eigs:ARPACKroutineErrorMinus14 ")
923  /* decrease the accuracy */
924  opts.tol= opts.tol * 100;
925  /* set the number of used Lanczos vectors */
926  opts.p= 50;
927  else
928  /* other errors should be passed back */
929  rethrow(err2);
930  end
931  end
932  end
933  else
934  /* other errors should be passed back */
935  rethrow(err);
936  end
937  end
938 
939  /* if we found an eigenvalue that looks like it's zero, it probably is
940  * zero, so ... */
941  if abs(lm) < sqrt(eps)
942  lm = 0;
943  end
944

946  warning(origstate);

974  function prepare() {
975 
976 
977  /* Create local copies of the truth solver structures. */
978  obj.normX= obj.solver.TrNorm;
979  /* more truth solver structures, this time only needed for the preparation */
980  normY = obj.solver.TeNorm;
981  affineB = obj.solver.Lhs;
982 
983  /* set the total number of needed addends for the affine representation */
984  obj.nQt= obj.solver.nQb * (obj.solver.nQb + 1) / 2;
985 
986  /* create the needed structure */
987  obj.affineT= cell(obj.nQt, 1);
988 
989  /* iterate over the different addends of the affine representation. as we
990  * are using an alternative expansion of the field independent operators to
991  * ensure that they are all symmetric and positiv semi-definite, we have to
992  * deal with the two following types of addends:
993  * 1. <T_q u, T_q u>_Y for q = 1..Qb,
994  * 2. <T_q u + T_p u, T_q u + T_p u>_Y for q = 1..Qb and p = (q+1)..Qb. */
995 
996  /* index for the affineT structure (a mapping of the index pair (q, p) to a
997  * useful 1d index would be nice, but... meh). */
998  idx = 1;
999 
1000  /* assemble the affine addends of the first kind */
1001  for qdx = 1:obj.solver.nQb
1002  obj.affineT[idx] = affineB[qdx].^t * (normY \ affineB[qdx]);
1003  idx = idx + 1;
1004  end
1005 
1006  /* and now assemble the affine addends of the second kind */
1007  for qdx = 1:obj.solver.nQb
1008  for pdx = (qdx + 1):obj.solver.nQb
1009  obj.affineT[idx] = (affineB[qdx] + affineB[pdx]).^t * ...
1010  (normY \ (affineB[qdx] + affineB[pdx]));
1011  idx = idx + 1;
1012  end
1013  end
1014 
1015  /* reset the greedy loop counter */
1016  obj.totalLoopCtr= 0;
1017  }
