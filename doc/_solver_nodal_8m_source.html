170  function matrixM = spacetimeStiffnessMatrix() {
171 
172 
173  /* temporal mass matrices */
174  MtAT = obj.temporal.massMatrix(" both ", obj.pd.tspan);
175  /* temporal "half stiffness" matrix */
176  CtAT = obj.temporal.halfStiffnessMatrix;
177 
178  /* temporal propagation vectors. this and the sign in front of the time
179  * derivative "half stiffness" matrix is the only difference between the
180  * forward and the backward propagator! */
181  if obj.isForward
182  et = obj.temporal.forwardInitVector;
183  tdsign = 1;
184  else
185  et = obj.temporal.backwardInitVector;
186  tdsign = -1;
187  end
188 
189  /* spatial mass matrices */
190  MxAT = obj.spatial.massMatrix;
191  MxATIc = obj.spatial.massMatrix;
192  /* spatial stiffness matrices */
193  AxAT = obj.spatial.stiffnessMatrix;
194 
195  /* compute the parts of the space time variational form as kronecker products */
196  timeDerivate = kron(CtAT, MxAT);
197  laplacian = kron(MtAT, AxAT);
198  offset = kron(MtAT, MxAT);
199  initialCond = kron(et, MxATIc);
200 
201  /* set the upper block of the system matrix */
202  M = tdsign * timeDerivate + obj.pd.laplacian * laplacian + obj.pd.offset * offset;
203 
204  /* resize it, so that we can add the lower block */
205  M(obj.nTestDim, obj.nTrialDim) = 0;
206 
207  /* add the lower block which is responsible for the propagation of the
208  * initial condition */
209  M((obj.temporal.nTest * obj.spatial.nTest + 1):end, :) = initialCond;
210  }

297  function colvecF = spacetimeLoadVector() {
298 
299 
300  /* set up the vector */
301  F = sparse(obj.nTestDim, 1);
302 
303  /* first we check whether the source term is a constant, only space
304  * dependent or depends on both time an space */
305  sourceIsConst = isnumeric(obj.pd.sourcefun);
306  if ~sourceIsConst
307  sourceIsSpatial = nargin(obj.pd.sourcefun) == 1;
308  else
309  sourceIsSpatial = false;
310  end
311 
312  /* compute the source term part of the load vector */
313  if sourceIsConst
314  /* first case: source is a constant value, so we can save some compuations */
315  g = zeros(obj.spatial.nTest, 1);
316  for jdx = 1:obj.spatial.nTest
317  g(jdx) = obj.spatial.evaluateFunctional(@(x) obj.pd.sourcefun, jdx);
318  end
319 
320  for kdx = 1:obj.temporal.nTest
321  /* calculate the temporal stepsize and save the calculated values */
322  delta_t = obj.pd.tgrid(kdx + 1) - obj.pd.tgrid(kdx);
323  pos = (kdx - 1) * obj.spatial.nTest + 1;
324  F(pos:(pos + obj.spatial.nTest - 1)) = delta_t * g;
325  end
326  elseif sourceIsSpatial
327  /* second case: source is a space only function, so we have the same for
328  * every point in time. */
329  g = zeros(obj.spatial.nTest, 1);
330  for jdx = 1:obj.spatial.nTest
331  g(jdx) = obj.spatial.evaluateFunctional(obj.pd.sourcefun, jdx);
332  end
333 
334  /* calculate the source term part */
335  for kdx = 1:obj.temporal.nTest
336  /* calculate the temporal stepsize and save the calculated values */
337  delta_t = obj.pd.tgrid(kdx + 1) - obj.pd.tgrid(kdx);
338  pos = (kdx - 1) * obj.spatial.nTest + 1;
339  F(pos:(pos + obj.spatial.nTest - 1)) = delta_t * g;
340  end
341  else
342  /* third case: source is both time and space dependent, so we use a
343  * trapezoidal rule for the temporal integral and matlabs numerical
344  * quadrature for the spatial integral. */
345 
346  /* precompute the needed functions for the trapezoidal rule */
347  gfun = cell(obj.temporal.nTrial, 1);
348  for kdx = 1:obj.temporal.nTrial
349  gfun[kdx] = @(x) obj.pd.sourcefun(obj.pd.tgrid(kdx), x);
350  end
351 
352  /* calculate the source term part */
353  for kdx = 1:obj.temporal.nTest
354  g = zeros(obj.spatial.nTest, 1);
355  for jdx = 1:obj.spatial.nTest
356  g(jdx) = obj.spatial.evaluateFunctional(@(x) (gfun[kdx + 1](x) + gfun[kdx](x)) / 2, jdx);
357  end
358 
359  /* calculate the temporal stepsize and save the calculated values */
360  delta_t = obj.pd.tgrid(kdx + 1) - obj.pd.tgrid(kdx);
361  pos = (kdx - 1) * obj.spatial.nTest + 1;
362  F(pos:(pos + obj.spatial.nTest - 1)) = delta_t * g;
363  end
364  end
365 
366  /* and now the same for the initial conditions */
367  icIsConst = isnumeric(obj.pd.icfun);
368 
369  /* the only difference in the computation is the way we forward the initial
370  * conditions to the spatial assembly object. */
371  if icIsConst
372  u0 = zeros(obj.spatial.nTest, 1);
373  for idx = 1:obj.spatial.nTest
374  u0(idx) = obj.spatial.evaluateFunctional(@(x) obj.pd.icfun, idx);
375  end
376  F(obj.spatial.nTest * obj.temporal.nTest + 1:end) = u0;
377  else
378  u0 = zeros(obj.spatial.nTest, 1);
379  for idx = 1:obj.spatial.nTest
380  u0(idx) = obj.spatial.evaluateFunctional(obj.pd.icfun, idx);
381  end
382  F(obj.spatial.nTest * obj.temporal.nTest + 1:end) = u0;
383  end
384 
385  /* also save the space time vector */
386  obj.Rhs= F;
387  }

480  function colvecx = lsqr(matrix A,colvec b) {
481 
482 
483  /* the algorithm is taken as-is from the above mentioned paper */
484  d = 0;
485  [vh, v, beta_] = obj.lsqrNormalize(b, obj.TeNorm);
486  [wh, w, alpha_] = obj.lsqrNormalize(A.^t * vh, obj.TrNorm);
487  rho_ = sqrt(beta_^2 + alpha_^2);
488  u = 0;
489  delta_ = alpha_;
490  gamma_ = beta_;
491 
492  /* start iterative procedure */
493  for iter = 1:550
494  d = wh - (alpha_ * beta_ / rho_^2) * d;
495  [vh, v, beta_] = obj.lsqrNormalize(A*wh - alpha_ * v, obj.TeNorm);
496  [wh, w, alpha_] = obj.lsqrNormalize(A.^t * vh - beta_ * w, obj.TrNorm);
497  rho_ = sqrt(beta_^2 + delta_^2);
498  u = u + (delta_ * gamma_ / rho_^2) * d;
499  delta_ = - delta_ * alpha_ / rho_;
500  gamma_ = gamma_ * beta_ / rho_;
501 
502  /* tolerance check */
503  if abs(delta_) * gamma_ < 1e-16
504  break;
505  end
506  end
507 
508  x = u;
509  }
