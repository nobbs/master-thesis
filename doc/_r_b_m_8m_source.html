540  function [doublebeta , doublegamma ] = calcDiscreteInfSupAndContinuityRB(matrix param) {
541 
542 
543  testshots = obj.assembleTestSnapshots(param);
544 
545  /* first we have to construct the rb system matrix for this parameter and
546  * the y- and x-norm matrices. */
547  Lhs = full(testshots.^t * obj.solver.spacetimeSystemMatrix(param) * obj.trialSnapshots);
548  Ynorm = full(testshots.^t * obj.solver.TeNorm * testshots);
549  Xnorm = full(obj.trialSnapshots.^t * obj.solver.TrNorm * obj.trialSnapshots);
550 
551  /* now we solve the generalized eigenvalue problem, the inf-sup and continuity
552  * constants are then the square roots of the smallest respectively largest
553  * generalized eigenvalue */
554  supNorm = Lhs.^t * (Ynorm \ Lhs);
555  ev = eig(supNorm, Xnorm);
556  beta = sqrt(min(ev));
557  gamma = sqrt(max(ev));
558  cond_ = gamma / beta
559  }
